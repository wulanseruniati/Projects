namespace Kucing
{
    public class Cat
    {
        string? color;
        string? species;
        int age;

        public string? favoriteFood;
        public string? catName;

        public Cat(string favoriteFood, string catName)
        {
            this.favoriteFood = favoriteFood;
            this.catName = catName;
        }

        public void Eat()
        {
            Console.WriteLine(catName + " eat " + favoriteFood);
        }
    }
}
